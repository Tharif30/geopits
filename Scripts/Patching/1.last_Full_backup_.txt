SELECT  
    bs.database_name,  
    bmf.physical_device_name AS BackupFile,  
    bs.backup_start_date,  
    bs.backup_finish_date,  
    DATEDIFF(MINUTE, bs.backup_start_date, bs.backup_finish_date) AS Duration_Minutes,  
    CAST(bs.backup_size / 1024 / 1024 AS DECIMAL(10,2)) AS BackupSize_MB,  
    CAST(bs.compressed_backup_size / 1024 / 1024 AS DECIMAL(10,2)) AS CompressedSize_MB  
FROM msdb.dbo.backupset bs  
JOIN msdb.dbo.backupmediafamily bmf  
    ON bs.media_set_id = bmf.media_set_id  
WHERE bs.type = 'D'  -- D = Full backup  
  AND bs.backup_finish_date = (  
        SELECT MAX(backup_finish_date)  
        FROM msdb.dbo.backupset  
        WHERE database_name = bs.database_name  
          AND type = 'D'  
    )  
ORDER BY bs.database_name;
